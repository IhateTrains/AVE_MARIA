		only_playable = yes

		ai = no
		ai_target_filter = self

			has_dlc = "Rajas of India"
			is_playable = yes

			holy_order = no

			}

			# The Indian religions can convert to each other anyway
			has_dharmic_religion_trigger = no
			capital_scope = { has_dharmic_religion_trigger = no }

			# May instead renounce iconoclasm
			NAND = {
				has_dlc = "Charlemagne"
				year <= 899
				has_landed_title = e_byzantium
				has_landed_title = c_byzantion
				religion = iconoclast
				capital_scope = { religion = orthodox }
			}
		}

		allow = {
			NOT = { trait = zealous }
			capital_scope = { ROOTs_religion_is_playable_trigger = yes }
			trigger_if = {
				limit = { capital_scope = { religion_group = ROOT } }
				prestige = 500
			}
			trigger_else = {
				prestige = 1000
			}
		}

		effect = {
			piety = 250
			if = {
				limit = {
					capital_scope = { religion_group = ROOT }
				}
				prestige = -500
			}
			else = {
				prestige = -1000
			}
			if = {
				limit = {
					NOR = {
						capital_scope = { religion_group = ROOT }
						trait = sympathy_christendom
						trait = sympathy_islam
						trait = sympathy_indian
						trait = sympathy_judaism
						trait = sympathy_pagans
						trait = sympathy_zoroastrianism
					}
				}
				random = {
					chance = 60
					trigger_switch = {
						on_trigger = religion_group

						christian = { add_trait = sympathy_christendom }
						muslim = { add_trait = sympathy_islam }
						indian_group = { add_trait = sympathy_indian }
						jewish_group = { add_trait = sympathy_judaism }
						pagan_group = { add_trait = sympathy_pagans }
						zoroastrian_group = { add_trait = sympathy_zoroastrianism }
					}
				}
			}
			custom_tooltip = { text = local_religion_conversion_tooltip }
			religion_authority = {
				modifier = ruler_converted_from
			}
			capital_scope = {
				religion_authority = {
					modifier = ruler_converted_to
				}
			}
			hidden_tooltip = {
				capital_scope = {
					reverse_religion = ROOT
				}
			}
		}

		ai_will_do = {
			factor = 0
		}

	}
